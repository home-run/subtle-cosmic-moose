55 /*
56  QMap uses qMapLessThanKey() to compare keys. The default
57  implementation uses operator<(). For pointer types,
58  qMapLessThanKey() casts the pointers to integers before it
59  compares them, because operator<() is undefined on pointers
60  that come from different memory blocks. (In practice, this
61  is only a problem when running a program such as
62  BoundsChecker.)
63 */
64 
65 template <class Key> inline bool qMapLessThanKey(const Key &key1, const Key &key2)
66 {
67  return key1 < key2;
68 }
69 
70 template <class Ptr> inline bool qMapLessThanKey(const Ptr *key1, const Ptr *key2)
71 {
72  Q_STATIC_ASSERT(sizeof(quintptr) == sizeof(const Ptr *));
73  return quintptr(key1) < quintptr(key2);
74 }

295 template <class Key, class T>
296 void QMapData<Key, T>::nodeRange(const Key &akey, QMapNode<Key, T> **firstNode, QMapNode<Key, T> **lastNode)
297 {
298  Node *n = root();
299  Node *l = end();
300  while (n) {
301  if (qMapLessThanKey(akey, n->key)) {
302  l = n;
303  n = n->leftNode();
304  } else if (qMapLessThanKey(n->key, akey)) {
305  n = n->rightNode();
306  } else {
307  *firstNode = n->leftNode() ? n->leftNode()->lowerBound(akey) : 0;
308  if (!*firstNode)
309  *firstNode = n;
310  *lastNode = n->rightNode() ? n->rightNode()->upperBound(akey) : 0;
311  if (!*lastNode)
312  *lastNode = l;
313  return;
314  }
315  }
316  *firstNode = *lastNode = l;
317 }
