424 template <typename T>
425 Q_INLINE_TEMPLATE void QList<T>::node_copy(Node *from, Node *to, Node *src)
426 {
427  Node *current = from;
428  if (QTypeInfo<T>::isLarge || QTypeInfo<T>::isStatic) {
429  QT_TRY {
430  while(current != to) {
431  current->v = new T(*reinterpret_cast<T*>(src->v));
432  ++current;
433  ++src;
434  }
435  } QT_CATCH(...) {
436  while (current-- != from)
437  delete reinterpret_cast<T*>(current->v);
438  QT_RETHROW;
439  }
440 
441  } else if (QTypeInfo<T>::isComplex) {
442  QT_TRY {
443  while(current != to) {
444  new (current) T(*reinterpret_cast<T*>(src));
445  ++current;
446  ++src;
447  }
448  } QT_CATCH(...) {
449  while (current-- != from)
450  (reinterpret_cast<T*>(current))->~T();
451  QT_RETHROW;
452  }
453  } else {
454  if (src != from && to - from > 0)
455  memcpy(from, src, (to - from) * sizeof(Node));
