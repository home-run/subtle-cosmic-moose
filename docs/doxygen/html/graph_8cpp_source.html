231 void Graph::shortestPath(Vertex source)
232 {
233  Heap<Vertex, vertexComp> vertexPQ;// Min-Heap (priority queue) containing all the
234  // vertices in the graph, ordered by the weight. Smallest
235  // weight is root
236  VertexSet T; // Contains the set of vertices to which the shortest path
237  // has been found.
238  Vertex u;
239  Vertex v;
240  Edge adjEdge;
241  int distanceSum;
242 
243  for(int i = 0; i < numVertices; i++)
244  {
245  vertexList[i].setDistance(0);
246  }
247 
248  source = vertexList.at(source.getId());
249  // Initialize all edges, and vertices to infinity
250  initialize_single_source(source);
251 
252  T.clear();
253  // Initialize the priority queue to start with the given source vertex as the first
254  // vertex to explore.
255  vertexPQ.insert(vertexList[source.getId()]);
256 
257  while(!vertexPQ.isEmpty())
258  {
259  u = vertexPQ.removeMin();
260  // While the current vertex u has an adjacent edge available.
261  while(vertexList.at(u.getId()).hasEdges() )
262  {
263  // Get the adjacent edge to the vertex. It will be the edge with the shortest
264  // path currently available
265  adjEdge = vertexList[u.getId()].getNearestEdge();
266  v = vertexList.at(adjEdge.idTo);
267  // checks to see if a path between vertex u and vertex v exists. A path
268  // exists if the value in the adjacency matrix is not 0
269  if(adjacencyMatrix[u.getId()][adjEdge.idTo] != 0 )
270  {
271  // Calculate the total distance between the vertex u and the weight
272  // between the vertex u and vertex v
273  distanceSum = u.getDistance() + adjacencyMatrix[u.getId()][v.getId()];
274 
275  // If the current distance to vertex v is greater than the sum of u's
276  // current distance plus the weight in the adj matrix set the distance
277  // to v to the new distance sum because it is a shorter path to that
278  // vertex.
279  if(v.getDistance() > distanceSum )
280  {
281  v.setDistance(distanceSum);
282  v.setParent(u.getId());
283  vertexList[v.getId()] = v;
284  }
285 
286  // If the set T of found vertices does not contain the vertex v add the
287  // vertex v to the priority queue to see the shorter path.
288  if(!T.contains(v))
289  {
290  vertexPQ.insert(v);
291  }
292  }
293  }
294  }
295  // Once the algorithm is complete, clear the VertexSet T of all vertices.
296  T.clear();
297 }
